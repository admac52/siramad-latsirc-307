float DETA(float MA[100][100], int m1){
   float deta=0;
   
	if(m1==1){
		deta=MA[0][0];
	}else{
		for(int i=0;i<m1;i++){
			deta=deta + ((MA[0][i])*(COFA(MA, m1, 0, i)));
		}
	}
   return deta;
}

float COFA(float MA[100][100], int m1, int fa, int ca){
   float AUXA[100][100],cofa;
   int rna=m1-1;
   int aa=0,ba=0;
   
   for(int i=0;i<m1;i++){
		for(int j=0;j<m1;j++){
			if((i!=fa) && (j!=ca)){
            	AUXA[aa][ba]=MA[i][j];
            	ba=ba+1;
	            if(ba>=rna){
	               aa=aa+1;
	               ba=0;
	            }
        	}
		}
	}
	cofa=pow(-1,(fa+ca)) * (DETA(AUXA, rna));
   return cofa;
}

float DETB(float MB[100][100], int m2){
   float detb=0;
   
	if(m2==1){

		for(int i=0;i<m2;i++){
			detb=detb + ((MB[0][i])*(COFB(MB, m2, 0, i)));
		}
	}
   return detb;
}

			if((i!=fb) && (j!=cb)){
            	AUXB[ab][bb]=MB[i][j];
            	bb=bb+1;
	            if(bb>=rnb){
	               ab=ab+1;
	               bb=0;
	            }
        	}
		}
	}
	cofb=pow(-1,(fb+cb)) * (DETB(AUXB, rnb));
	
   return cofb;
